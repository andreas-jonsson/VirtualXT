55 #define vxtu_randomize(ptr, size, seed) { \
56  int s = (int)(seed); \
57  for (int i = 0; i < (int)(size); i++) { \
58  const int m = 2147483647; \
59  s = (16807 * s) % m; \
60  float r = (float)s / m; \
61  ((vxt_byte*)(ptr))[i] = (vxt_byte)(255.0 * r); \
62  } \
63 } \

85 #define vxtu_static_allocator(name, size) \
86  static vxt_byte name ## allocator_data[(size)]; \
87  static vxt_byte * name ## allocator_ptr = name ## allocator_data; \
88  \
89  void * name (void *ptr, size_t sz) { \
90  if (!sz) \
91  return NULL; \
92  else if (ptr) /* Reallocation is not support! */ \
93  return NULL; \
94  \
95  vxt_byte *p = name ## allocator_ptr; \
96  sz += 8 - (sz % 8); \
97  name ## allocator_ptr += sz; \
98  \
99  if (name ## allocator_ptr >= (name ## allocator_data + (size))) \
100  return NULL; /* Allocator is out of memory! */ \
101  return p; \
102  } \
