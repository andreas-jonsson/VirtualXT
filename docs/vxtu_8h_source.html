48 #define vxtu_static_allocator(name, size) \
49  static vxt_byte name ## allocator_data[(size)]; \
50  static vxt_byte * name ## allocator_ptr = name ## allocator_data; \
51  \
52  void * name (void *ptr, size_t sz) { \
53  if (!sz) \
54  return NULL; \
55  else if (ptr) /* Reallocation is not support! */ \
56  return NULL; \
57  \
58  vxt_byte *p = name ## allocator_ptr; \
59  sz += 8 - (sz % 8); \
60  name ## allocator_ptr += sz; \
61  \
62  if (name ## allocator_ptr >= (name ## allocator_data + (size))) \
63  return NULL; /* Allocator is out of memory! */ \
64  return p; \
65  } \

71  static vxt_byte *vxtu_read_file(vxt_allocator *alloc, const char *file, int *size) {
72  vxt_byte *data = NULL;
73  FILE *fp = fopen(file, "rb");
74  if (!fp)
75  return data;
76 
77  if (fseek(fp, 0, SEEK_END))
78  goto error;
79 
80  int sz = (int)ftell(fp);
81  if (size)
82  *size = sz;
83 
84  if (fseek(fp, 0, SEEK_SET))
85  goto error;
86 
87  data = (vxt_byte*)alloc(NULL, sz);
88  if (!data)
89  goto error;
90 
91  memset(data, 0, sz);
92  if (fread(data, 1, sz, fp) != (size_t)sz) {
93  alloc(data, 0);
94  goto error;
95  }
96 
97  error:
98  fclose(fp);
99  return data;
100  }
