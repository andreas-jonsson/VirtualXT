135  static vxt_byte *vxtu_read_file(vxt_allocator *alloc, const char *file, int *size) {
136  vxt_byte *data = NULL;
137  FILE *fp = fopen(file, "rb");
138  if (!fp)
139  return data;
140 
141  if (fseek(fp, 0, SEEK_END))
142  goto error;
143 
144  int sz = (int)ftell(fp);
145  if (size)
146  *size = sz;
147 
148  if (fseek(fp, 0, SEEK_SET))
149  goto error;
150 
151  data = (vxt_byte*)alloc(NULL, sz);
152  if (!data)
153  goto error;
154 
155  memset(data, 0, sz);
156  if (fread(data, 1, sz, fp) != (size_t)sz) {
157  alloc(data, 0);
158  goto error;
159  }
160 
161  error:
162  fclose(fp);
163  return data;
164  }
