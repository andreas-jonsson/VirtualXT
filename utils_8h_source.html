38  static vxt_byte *vxtu_read_file(vxt_allocator *alloc, const char *file, int *size) {
39  vxt_byte *data = NULL;
40  FILE *fp = fopen(file, "rb");
41  if (!fp)
42  return data;
43 
44  if (fseek(fp, 0, SEEK_END))
45  goto error;
46 
47  int sz = (int)ftell(fp);
48  if (size)
49  *size = sz;
50 
51  if (fseek(fp, 0, SEEK_SET))
52  goto error;
53 
54  data = (vxt_byte*)alloc(NULL, sz);
55  if (!data)
56  goto error;
57 
58  memset(data, 0, sz);
59  if (fread(data, 1, sz, fp) != (size_t)sz) {
60  alloc(data, 0);
61  goto error;
62  }
63 
64  error:
65  fclose(fp);
66  return data;
67  }
